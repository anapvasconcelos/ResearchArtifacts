\newcommand{\ShowPercentageNoHundred}[2]{%
    \edef\percentage{\fpeval{round(#1/#2*100,1)}}%
    \ifdim\percentage pt < 10pt
        (\hspace*{4.3pt}\num{\percentage}\%)
    \else
        (\num{\percentage}\%)
    \fi
}

\newcommand*\ChartWithAbsoluteAndPercentageNumbers[2]{
    \numprint{#1}&
    \ShowPercentageNoHundred{#1}{#2}&
    \raisebox{-0.05em}{\inlinechart{#1}{#2}}
}

    \textcolor{black!80}{\rule{\blackbar\BARSIZE}{1.6ex}}%
    \textcolor{black!10}{\rule{\whitebar\BARSIZE}{1.6ex}}%
}
